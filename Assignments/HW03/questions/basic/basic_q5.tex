 موارد خواسته شده را انجام دهید:
 
 
 \begin{enumerate}
 	\item 
 	با استفاده از جدول کارنو، تابع زیر را ساده کنید و سپس مدار تمام \lr{NAND} تابع ساده شده را رسم کنید.
 	\begin{latin}
	 	$f (a,b,c,d) = a'\cdot b'\cdot c\cdot d' + a' \cdot b \cdot c' + b \cdot c \cdot d + a' \cdot c \cdot d' + b' \cdot c \cdot d' + a \cdot b' \cdot c$
 	\end{latin}
 	
 	
 	\item 
 	با استفاده از جدول کارنو، تابع زیر را ساده کنید و سپس مدار تمام \lr{NOR} تابع ساده شده را رسم کنید.
 	
 	\begin{latin}
 		$f (a,b,c,d) = a' \cdot b' \cdot c' + a' \cdot b' \cdot c + a \cdot b' + b' \cdot c' \cdot d + b' \cdot c \cdot d'$ 
 	\end{latin}
 	
 	
 \end{enumerate}
