موارد خواسته شده را به دست آورید.

\begin{enumerate}
	\item 
	توابع زیر را تاحدامکان ساده کنید و درنهایت مدار آن را رسم کنید.
	
	\begin{latin}
		\begin{enumerate}
			\item $F(A, B, C, D) = ((A + B').(C + D))'$
			
			\item $F(A,B,C,D) = [A + (BCD)' ][ (AD)' + B(C' + A) ]$
			
			\item $F(A,B,C,D,E,F,G,H) = [ (A + B)' . (C + D)' . (E + F)' . (G + H)' ]'$
		\end{enumerate}
	\end{latin}
	
	
	\item	
	درستی یا نادرستی برابری‌های زیر را با استفاده از جدول ارزش‌ها\LTRfootnote{Truth Table} ارزیابی کنید.
	\begin{latin}
		\begin{enumerate}
			% \item $A + A'B = A + B$
			
			\item $A \oplus B = (A.B') + (A'.B)$
			
			\item $(A \oplus B) . (C' + D) = (A . B' . C') + (B . A' . C') + ((A \oplus B) . D)$
			
			\item $ABC′ + AB'C + B'C'D + BCD = AB'C + ABC' + AD + BCD + B'C'D$
		\end{enumerate}
	\end{latin}
	
	
	\item 
	عبارات (i) و (ii) را به‌فرم \lr{POS} و عبارت (iii) را به‌فرم \lr{SOP} بازنویسی کنید.
	\begin{latin}
		\begin{enumerate}
			\item $AB + CD (AB' + CD)$
			
			\item $AB (B' C' + BC)$
			
			\item $A + B[AC + (B + C)' D]$
		\end{enumerate}
	\end{latin}
	
	
	
	
\end{enumerate}
