\begin{latin}
	\begin{enumerate}
		\item 
		$[(AB)' + C'D]'$
		
		\item 
		$AB'C + (A' + B + D)(ABD' + B')$
		
		\item 
		$[ ((A + B)' . (C + D)' . (E + F)' . (G + H)')']'$
		
		\item 
		$(V' + U + W).[(W + X) + Y + UZ'] + [ (W + X) + UZ' + Y ]$
	\end{enumerate}
\end{latin}
