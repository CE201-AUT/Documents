توابع زیر را از فرم استاندارد به فرم کاننیکال تبدیل کنید:

\begin{latin}
	\begin{enumerate}
		\item 
		$ f(A, B, C) = A C' + B C' + A B C $
		
		\item 
		$ f(A, B, C) = (C + A')(B + C')(A + B + C)(A' + B') $
		
		\item 
		$ f(A, B, C, D, E) = B \cdot D' \cdot E + A \cdot B' \cdot D + A \cdot C \cdot D' \cdot E + A \cdot C' \cdot E $
	\end{enumerate}
\end{latin}
